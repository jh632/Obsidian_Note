# 嵌入式设计模式-工厂模式

## 概述

工厂模式（Factory Pattern）将**对象的创建逻辑**封装起来，调用者只需说"我要一个某某类型的驱动"，而不需要知道具体怎么 new/alloc、怎么初始化。在嵌入式 C 中，它的核心形式就是**一个函数根据参数返回不同的 ops 结构体指针**。

## 核心概念

- **产品（Product）**：创建出来的对象，通常是 `xxx_handle_t` 或 `xxx_ops_t *`
- **创建器（Creator / Factory）**：封装创建逻辑的函数或模块
- **两种变体**：
  - **工厂方法（Factory Method）**：一个工厂函数，根据参数返回不同实现
  - **抽象工厂（Abstract Factory）**：一组工厂函数，创建一族配套的产品

## 细节

### 为什么嵌入式需要工厂模式

嵌入式 C 中没有构造函数，对象的创建是手动完成的——你写 `i2c_master_bus_add_device()` 而不是 `new I2cDevice()`。工厂模式把这种手动创建逻辑收拢到一处，带来三个好处：

| 问题 | 没有工厂 | 有工厂 |
|---|---|---|
| 新增驱动型号 | 在所有调用处加 if-else | 工厂内部加一条 entry |
| 驱动初始化步骤变化 | 每个调用处都得改 | 工厂统一改 |
| 调用者需要知道多少信息 | 必须知道全部硬件细节 | 只需知道 chip_id / 设备名 |

### 1. 工厂方法模式（Factory Method）

最常用的形式：**输入参数 → 返回对应的 ops 结构体**。

```c
// ============ 产品接口 ============
typedef struct {
    esp_err_t (*init)(void *hw_cfg, uint8_t addr);
    esp_err_t (*read_temp)(void *hw_cfg, int32_t *temp);
} temp_sensor_ops_t;

// ============ 具体产品 A：DS18B20（OneWire） ============
static esp_err_t ds18b20_init(void *cfg, uint8_t addr) { ... }
static esp_err_t ds18b20_read(void *cfg, int32_t *temp) { ... }
temp_sensor_ops_t ds18b20_ops = {
    .init      = ds18b20_init,
    .read_temp = ds18b20_read,
};

// ============ 具体产品 B：SHT30（I2C） ============
static esp_err_t sht30_init(void *cfg, uint8_t addr) { ... }
static esp_err_t sht30_read(void *cfg, int32_t *temp) { ... }
temp_sensor_ops_t sht30_ops = {
    .init      = sht30_init,
    .read_temp = sht30_read,
};

// ============ 工厂 ============
temp_sensor_ops_t *temp_sensor_factory(const char *chip_name) {
    if (strcmp(chip_name, "DS18B20") == 0) return &ds18b20_ops;
    if (strcmp(chip_name, "SHT30")   == 0) return &sht30_ops;
    return NULL;  // 不支持的型号
}
```

**配合桥接模式**：工厂负责装配 handle 的两端。

```c
// 工厂 + 桥接 —— 完整的驱动创建过程
typedef struct {
    temp_sensor_ops_t *ops;   // 桥接：实现层
    void              *cfg;   // 桥接：实现层配置
    int32_t            scale; // 抽象层自有状态
} temp_sensor_t;

temp_sensor_t *temp_sensor_create(const char *chip_name, void *hw_cfg) {
    temp_sensor_ops_t *ops = temp_sensor_factory(chip_name);
    if (!ops) return NULL;

    temp_sensor_t *sensor = malloc(sizeof(temp_sensor_t));
    if (!sensor) return NULL;

    sensor->ops   = ops;
    sensor->cfg   = hw_cfg;
    sensor->scale = 100;  // 抽象层默认值

    if (ops->init(hw_cfg, 0x00) != ESP_OK) {
        free(sensor);
        return NULL;
    }
    return sensor;
}
```

### 2. 抽象工厂模式（Abstract Factory）

当需要创建**一族相关的资源**时，用抽象工厂。它返回的不是一个 ops，而是一组工厂函数。

```c
// ============ 抽象工厂接口 ============
typedef struct {
    i2c_master_bus_handle_t (*create_bus)(i2c_config_t *cfg);
    spi_device_handle_t     (*create_spi)(spi_config_t *cfg);
    gpio_num_t              (*get_led_pin)(void);
} bsp_factory_t;

// ============ 具体工厂 A：ESP32-C3 ============
static i2c_master_bus_handle_t c3_create_i2c(i2c_config_t *cfg) {
    cfg->scl_io_num = GPIO_NUM_5;   // C3 的固定管脚映射
    cfg->sda_io_num = GPIO_NUM_4;
    return i2c_master_bus_add(cfg);
}
static spi_device_handle_t c3_create_spi(spi_config_t *cfg) {
    cfg->mosi = GPIO_NUM_7;
    ...
}
static gpio_num_t c3_get_led(void) { return GPIO_NUM_8; }

bsp_factory_t esp32c3_factory = {
    .create_bus = c3_create_i2c,
    .create_spi = c3_create_spi,
    .get_led_pin = c3_get_led,
};

// ============ 具体工厂 B：ESP32-S3 ============
bsp_factory_t esp32s3_factory = {
    .create_bus = s3_create_i2c,   // S3 管脚不同
    .create_spi = s3_create_spi,
    .get_led_pin = s3_get_led,
};

// ============ 使用 ============
bsp_factory_t *bsp = detect_platform() == CHIP_C3 ? &esp32c3_factory : &esp32s3_factory;
i2c_master_bus_handle_t bus = bsp->create_bus(&user_cfg);  // 不需要知道管脚
```

### 3. 驱动注册表模式（嵌入式中最常见的变体）

嵌入式工厂最常用的形式不是显式调用 `factory()`，而是**驱动自注册 + 全局查找表**。

```c
// ============ 驱动注册表 ============
#define MAX_DRIVERS  16

typedef struct {
    const char         *name;
    temp_sensor_ops_t  *ops;
} driver_entry_t;

static driver_entry_t s_drivers[MAX_DRIVERS];
static int s_drv_count = 0;

esp_err_t driver_register(const char *name, temp_sensor_ops_t *ops) {
    if (s_drv_count >= MAX_DRIVERS) return ESP_ERR_NO_MEM;
    s_drivers[s_drv_count].name = name;
    s_drivers[s_drv_count].ops  = ops;
    s_drv_count++;
    return ESP_OK;
}

temp_sensor_ops_t *driver_find(const char *name) {
    for (int i = 0; i < s_drv_count; i++) {
        if (strcmp(s_drivers[i].name, name) == 0)
            return s_drivers[i].ops;
    }
    return NULL;
}

// ============ 驱动自注册（编译期隐式调用） ============
// ds18b20_driver.c
static void auto_register(void) {
    driver_register("DS18B20", &ds18b20_ops);
}
// GCC: 在 init 阶段自动执行
__attribute__((constructor)) static void ds18b20_auto_init(void) { auto_register(); }

// sht30_driver.c
__attribute__((constructor)) static void sht30_auto_init(void) {
    driver_register("SHT30", &sht30_ops);
}

// ============ 使用方：完全不知道具体型号 ============
void app_main(void) {
    // 编译链接了多少个驱动，register 表里就有多少
    temp_sensor_ops_t *ops = driver_find("DS18B20");
    // 使用 ops ...
}
```

> **自注册的好处**：新建一个驱动 .c 文件，自动注册到全局表，**不需要修改任何工厂代码**。这是嵌入式工程中实现"插件化"的核心手法。

### 在 ESP-IDF 中的体现

| API | 模式 | 说明 |
|---|---|---|
| `i2c_master_bus_add_device()` | 工厂方法 | 传入配置，返回设备句柄 |
| `spi_bus_add_device()` | 工厂方法 | 同上，传入配置返回设备句柄 |
| `adc_oneshot_new_unit()` | 工厂方法 | 传入 adc_cfg，返回 handle |
| `esp_efuse_get_field()` | 工厂（间接） | 内部根据芯片型号选择不同 efuse 布局 |
| 各 BSP 的 `bsp_init_*()` | 抽象工厂 | 按开发板提供一组外设初始化函数 |

### 工厂 vs 桥接：一对共生模式

```c
// 工厂 + 桥接 的完整协作流程
// 桥接定了接口（ops + handle 结构）
// 工厂负责在运行时把两端装配起来

temp_sensor_t *sensor = temp_sensor_create("SHT30", &i2c_cfg);
//                              ↑ 工厂负责：选 ops + alloc handle + init
//                              ↑ 桥接负责：sensor->ops->read_temp() 不关心具体实现
```

| 模式 | 关注 | 时机 |
|---|---|---|
| 桥接 | 抽象与实现分离后如何交互 | 运行时调用 |
| 工厂 | 分离后的两端如何装配起来 | 创建时装配 |

**判断方法**：如果你在问"上层代码怎么调下层"——这是桥接。如果你在问"上层代码怎么拿到下层对象"——这是工厂。
